import * as XLSX from 'xlsx';

const aiInventoryFile = await window.fs.readFile('AIToolInventory.xlsx');
const workbook = XLSX.read(aiInventoryFile, {
    cellStyles: true,
    cellFormulas: true,
    cellDates: true
});

const sheet = workbook.Sheets[workbook.SheetNames[0]];
const data = XLSX.utils.sheet_to_json(sheet, { header: 1 });

// Generate INSERT statements for all AI tools (skip header row)
for (let i = 1; i < data.length; i++) {
  const row = data[i];
  if (!row[0]) continue; // Skip empty rows

  const cleanData = {
    title: (row[0] || 'Untitled AI Tool').replace(/'/g, "''"),
    description: (row[3] || 'No description provided').replace(/'/g, "''"),
    process: (row[4] || 'Unknown').replace(/'/g, "''"),
    aiOrModel: row[1] || 'AI',
    useCaseStatus: (row[2] || 'Active').replace(/'/g, "''"),
    riskToCustomers: (row[5] || '').replace(/'/g, "''"),
    riskToRsa: (row[6] || '').replace(/'/g, "''"),
    dataUsed: (row[7] || '').replace(/'/g, "''"),
    modelOwner: (row[8] || '').replace(/'/g, "''"),
    rsaPolicyGovernance: (row[9] || '').replace(/'/g, "''"),
    thirdPartyModel: (row[10] === 'Yes' || row[10] === 'Select') ? 'yes' : 'no',
    validationResponsibility: (row[11] || '').replace(/'/g, "''"),
    informedBy: (row[12] || '').replace(/'/g, "''")
  };

  // Execute INSERT for each AI tool
  await db.insert(useCases).values({
    title: cleanData.title,
    description: cleanData.description,
    process: cleanData.process,
    use_case_type: 'AI Tool',
    aiOrModel: cleanData.aiOrModel,
    useCaseStatus: cleanData.useCaseStatus,
    riskToCustomers: cleanData.riskToCustomers,
    riskToRsa: cleanData.riskToRsa,
    dataUsed: cleanData.dataUsed,
    modelOwner: cleanData.modelOwner,
    rsaPolicyGovernance: cleanData.rsaPolicyGovernance,
    thirdPartyModel: cleanData.thirdPartyModel,
    validationResponsibility: cleanData.validationResponsibility,
    informedBy: cleanData.informedBy,
    library_source: 'sharepoint_import',
    library_tier: 'reference',
    is_active_for_rsa: 'false',
    is_dashboard_visible: 'true',
    // Default strategic fields to 0 (not applicable for AI tools)
    revenue_impact: 0,
    cost_savings: 0,
    risk_reduction: 0,
    strategic_fit: 0,
    data_readiness: 0,
    technical_complexity: 0,
    change_impact: 0,
    adoption_readiness: 0,
    broker_partner_experience: 0,
    model_risk: 0,
    explainability_bias: 0,
    regulatory_compliance: 0
  });
}
